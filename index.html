15. 배열 조회하기


          {
            const arr1 = [100, 200, 300, 400, 500];
            const search = 300;
          
            document.write('*** 15. 배열 조회하기 ***');
          
            // for문
            for (let i = 0; i < arr1.length; i++) {
              if (search === arr1[i]) {
                document.write(search + '를 찾았습니다.');
              }
            }
            // forEach()
            arr1.forEach(function (element) {
              if (search === element) {
                document.write(search + '를 찾았습니다.');
              }
            });
            // map()
            arr1.map(function (element) {
              if (search === element) {
                document.write(search + '를 찾았습니다.');
              }
            });
            // for in문
            for (let i in arr1) {
              if (search === arr1[i]) {
                document.write(search + '를 찾았습니다.');
              }
            }
            // for of문
            for (let element of arr1) {
              if (search === element) {
                document.write(search + '를 찾았습니다.');
              }
            }
          }
